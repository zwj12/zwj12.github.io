# 安装typescript:
	npm install -g typescript

# 安装yarn:
	npm install -g yarn

# 创建React项目：
	npx create-react-app my-app
	npx create-react-app my-app --template typescript

# npm指令
	npm start
	npm run build

# 非根目录部署
如在机器人Home文件夹的docs子文件夹中部署，需要在package.json配置文件中添加homepage属性：

	{
  		"homepage": ".",
	}
