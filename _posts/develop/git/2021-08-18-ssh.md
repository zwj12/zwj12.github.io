# Git Bash
使用Git Bash运行命令行

# 查看是否已有SSH
	ls -al ~/.ssh

# 创建SSH
	ssh-keygen -t rsa -C "zwj_12@163.com"	


# 把SSH Key添加到ssh-agent中去
	eval  "ssh-agent -s"
	ssh-agent bash
	ssh-add ~/.ssh/id_rsa

# 测试SSH连接
	ssh -T git@github.com

# 切换远程仓库到SSH方式
	git remote -v
	git remote set-url origin git@github.com:zwj12/TpsViewZhongXun.git
	git remote -v
